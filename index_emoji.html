      tile.classList.add('tile');

      tile.dataset.index = i;

      tile.textContent = '❓';


      tile.addEventListener('click', () => {

        if (gameOver) return;


        if (parseInt(tile.dataset.index) === cakeIndex) {

        } else {

        }

      });


      grid.appendChild(tile);

    }
